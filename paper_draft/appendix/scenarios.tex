\section{Experimental Scenarios}
\label{app:scenarios}

\Tabref{tab:scenarios} lists the 20 scenarios used in Experiment~2, organized by category.

\begin{table}[h]
\centering
\caption{Full list of scenarios used in the theory distinguishability experiment (Experiment~2).}
\label{tab:scenarios}
\small
\begin{tabular}{@{}llp{7.5cm}@{}}
\toprule
\textbf{Category} & \textbf{Scenario} & \textbf{Description} \\
\midrule
\multirow{4}{*}{Neurological}
    & Split-brain & A patient whose corpus callosum has been severed, with each hemisphere operating independently \\
    & Blindsight & A patient with V1 damage who responds to visual stimuli without reporting awareness \\
    & Coma awareness & A patient in apparent coma who shows neural responses to spoken commands \\
    & Locked-in & A fully aware patient with complete motor paralysis except eye movement \\
\midrule
\multirow{4}{*}{Altered states}
    & Dreaming & A person in REM sleep experiencing vivid narrative dreams \\
    & Meditation & An experienced meditator in deep absorption with reduced self-referential processing \\
    & Sleepwalking & A person performing complex motor behaviors while electroencephalographically asleep \\
    & Anesthesia & A patient under general anesthesia with suppressed cortical activity \\
\midrule
\multirow{4}{*}{Non-human}
    & Octopus & An octopus solving a novel puzzle, displaying flexible problem-solving \\
    & Ant colony & An ant colony exhibiting collective decision-making and adaptive foraging \\
    & Cerebral organoid & A lab-grown brain organoid showing spontaneous oscillatory activity \\
    & Thermostat & A simple thermostat with a feedback loop maintaining temperature \\
\midrule
\multirow{3}{*}{AI / Simulation}
    & GPT conversation & A large language model producing coherent, contextually appropriate dialogue \\
    & Perfect simulation & A neuron-by-neuron simulation on feedforward lookup tables with identical I/O \\
    & Robot costume & A human-appearing robot that removes its costume, revealing machinery \\
\midrule
\multirow{5}{*}{Philosophical}
    & Zombie twin & A molecule-for-molecule duplicate stipulated to lack phenomenal experience \\
    & Gradual replacement & A brain whose neurons are replaced one-by-one with functional silicon equivalents \\
    & Hemisphere isolation & One hemisphere isolated and sustained independently \\
    & Binocular rivalry & A subject experiencing alternating perceptual interpretations of ambiguous input \\
    & Infant & A 3-month-old infant showing preferential looking and social engagement \\
\bottomrule
\end{tabular}
\end{table}
